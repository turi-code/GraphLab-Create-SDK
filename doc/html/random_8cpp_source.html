141  ASSERT_TRUE(tls_ptr != NULL);

219  generator& get_source() {
220  // get the thread local storage
221  generator* tls_rnd_ptr =
222  reinterpret_cast<generator*>
223  (pthread_getspecific(get_random_source_key()));
224  // Create a tls_random_source if none was provided
225  if(tls_rnd_ptr == NULL) {
226  tls_rnd_ptr = new generator();
227  assert(tls_rnd_ptr != NULL);
228  // This will seed it with the master rng
229  source_registry::global().register_generator(tls_rnd_ptr);
230  pthread_setspecific(get_random_source_key(),
231  tls_rnd_ptr);
232  }
233  // assert(tls_rnd_ptr != NULL);
234  return *tls_rnd_ptr;
